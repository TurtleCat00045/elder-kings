##### Specialization-locked spells
### If school is main skill of the specialization (using the classes from Morrowind) then it starts from spe_1 to spe_3
### If school is secondary skill of the specialization (same), then it starts from spe_2 to spe_3. So you can't cast master spells from a secondary school

### Alteration
can_use_alteration = {
	OR = {
		can_use_alteration_journeyman = yes
		can_use_alteration_expert = yes
		can_use_alteration_master = yes
	}
}

can_use_alteration_journeyman = {
	OR = {
		is_battlemage = yes
		is_healer = yes
		is_nightblade = yes
		is_priest_education = yes
		is_spellsword = yes
	}
}

can_use_alteration_expert = {
	OR = {
		trait = battlemage_2
		trait = battlemage_3
		trait = healer_2
		trait = healer_3
		trait = nightblade_2
		trait = nightblade_3
		trait = priest_2
		trait = priest_3
		trait = spellsword_2
		trait = spellsword_3
	}
}

can_use_alteration_master = {
	OR = {
		trait = battlemage_3
		trait = healer_3
		trait = nightblade_3
		trait = priest_3
		trait = spellsword_3
	}
}
